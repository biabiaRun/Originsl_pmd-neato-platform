ROYALE_INCLUDE_TARGET_FILE("${ROYALE_SOURCE_DIR}/cmake/flags")

if (ROYALE_LOGGING)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -DROYALE_USE_LOGGING")
endif ()

if (ROYALE_LOGGING_FORCE)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -DROYALE_FORCE_LOGGING -DROYALE_USE_LOGGING")
endif ()

if (ROYALE_ENABLE_TIME_MEASUREMENTS)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -DROYALE_FORCE_LOGGING -DROYALE_USE_LOGGING")
endif ()

if (ROYALE_LOGGING_SYSLOG)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -DROYALE_LOG_BACKEND_SYSLOG")
endif ()

if (ROYALE_LOGGING_WIN_DEBUG_STRING_LOG)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -DROYALE_LOGBACKEND_WIN_DEBUG_STRING_LOG")
endif ()
